\begin{algorithm}[t]
\caption{Temporally Grounded C-BELIEF Reasoning Pipeline}
\label{alg:cbelief_reasoning}
\begin{algorithmic}[1]
\Require Standardized renal evidence events $\mathcal{E}$ for one admission; query-time boundary policy
\Ensure Structured prediction with initial/final claim status, renal hypothesis, evidence support, and temporal-safety flags

\State Standardize raw renal evidence into events with identifiers, timestamps, source fields, visibility labels, and event text.
\State Sort events by patient, admission, timestamp, and event identifier to form a longitudinal event stream.
\State Select query time $t_q$ using the first creatinine trend event when available; otherwise use the earliest sufficient observed event.
\State Partition events into:
\[
\mathcal{E}_{vis}=\{e \in \mathcal{E}: time(e)\le t_q,\ visibility(e)=observed\},
\]
\[
\mathcal{E}_{future}=\{e \in \mathcal{E}: time(e)>t_q\},
\quad
\mathcal{E}_{retro}=\{e \in \mathcal{E}: visibility(e)\in retrospective\}.
\]

\If{using rule-based C-BELIEF}
    \State Extract visible creatinine trajectory features from $\mathcal{E}_{vis}$: minimum, maximum, first, last, deltas, ratios, and supporting event IDs.
    \State Detect visible AKI, CKD, dialysis/RRT, and renal text evidence.
    \State Score candidate hypotheses:
    \[
    h \in \{\textit{acute renal deterioration},\ \textit{chronic renal dysfunction},\ \textit{uncertain/transient abnormality}\}.
    \]
    \State Assign acute score for clear or low-confidence creatinine rise and visible AKI/RRT evidence.
    \State Assign chronic score for stable high creatinine and visible CKD evidence.
    \State Assign uncertain score when visible evidence is insufficient for acute or chronic disease.
    \State Select the primary initial hypothesis by maximum visible-evidence score.
    \State Map evidence strength to initial claim status: \textit{supported}, \textit{partially supported}, or \textit{insufficient}.
    \State Return belief distribution and visible supporting evidence IDs.
\Else
    \State Run \textsc{TemporalGatekeeper} to classify evidence as visible, future, retrospective, or temporally unsafe.
    \State Run \textsc{HypothesisPanel} to infer initial hypotheses using $\mathcal{E}_{vis}$ and final hypotheses using $\mathcal{E}_{vis}\cup\mathcal{E}_{future}\cup\mathcal{E}_{retro}$.
    \State Run \textsc{EvidenceAdjudicator} to select valid initial and final supporting evidence IDs.
    \State Run \textsc{DelayedReattribution} to determine whether future or retrospective evidence changes the final clinical phenotype.
    \State Run \textsc{ClinicalCoordinator} to produce the final structured prediction.
    \State Normalize prediction labels to the allowed schema.
\EndIf

\State Verify temporal safety deterministically: initial supporting evidence IDs must be contained in $\mathcal{E}_{vis}$.
\State Flag future leakage, retrospective misuse, unknown support IDs, and delayed reattribution when applicable.
\State \Return Structured prediction and temporal verification report.
\end{algorithmic}
\end{algorithm}

\begin{algorithm}[t]
\caption{AI-Assisted Quality Control and Label Refinement Workflow}
\label{alg:cbelief_qc}
\begin{algorithmic}[1]
\Require Full C-BELIEF sample set $\mathcal{D}$ with visible and future summaries; automatic labels; clinician audit policy
\Ensure Audited dataset version with accepted label corrections and field-level change log

\State Generate compact audit summaries for each sample:
\Statex \hspace{\algorithmicindent} visible summary for query-time judgment and future/retrospective summary for final judgment.
\State Apply automatic AI prescreening to all samples in $\mathcal{D}$.
\State Assign each sample an AI review status, suspected issue, review priority, and confidence.
\State Split prescreened samples into AI-pass and AI-flagged subsets.

\State Construct clinician review spreadsheets.
\Statex \hspace{\algorithmicindent} For initial judgment, reviewers may use only the visible summary.
\Statex \hspace{\algorithmicindent} For final judgment, reviewers may use visible plus future/retrospective summaries.
\State Collect clinician annotations for initial validity, final validity, evidence correctness, temporal validity, suspected issue, confidence, and comments.
\State Summarize clinician-confirmed errors among AI-flagged samples.
\State Estimate review yield $P(\text{confirmed error}\mid\text{AI flagged})$ and stratify errors by issue type and sample subtype.

\State Select high-risk AI-flagged cases:
\[
\{\textit{ambiguous AKI/CKD},\ \textit{wrong judgment},\ \textit{temporal leakage}\}.
\]
\State Run AI-assisted relabeling on selected high-risk cases with strict temporal label definitions:
\Statex \hspace{\algorithmicindent} initial labels must use visible evidence only;
\Statex \hspace{\algorithmicindent} final labels may use visible, future, and retrospective evidence;
\Statex \hspace{\algorithmicindent} valid delayed reattribution must not be counted as temporal leakage.
\State Parse model output into a fixed schema with corrected labels, error type, evidence basis, invalid-support flag, rationale, and confidence.
\State Validate each relabeling output against the allowed label vocabulary and schema.
\State Compare corrected labels with existing labels to identify actual field-level changes.

\For{each AI-assisted relabeling result}
    \If{schema is valid \textbf{and} decision is confirmed error \textbf{and} confidence $\geq 0.90$ \textbf{and} labels changed}
        \If{error type is temporal leakage}
            \State Auto-accept correction only if invalid future or retrospective initial support is explicitly identified.
        \Else
            \State Auto-accept correction.
        \EndIf
    \Else
        \State Route case to human adjudication or no-error archive.
    \EndIf
\EndFor

\State Apply accepted corrections to a new dataset version without overwriting the original dataset.
\State Update initial/final claim status, primary hypotheses, final clinical phenotype, and delayed reattribution fields.
\State Record correction provenance, error type, confidence, rationale, and all field-level changes.
\State \Return Corrected dataset, accepted-correction table, human-adjudication table, no-error table, and change log.
\end{algorithmic}
\end{algorithm}
